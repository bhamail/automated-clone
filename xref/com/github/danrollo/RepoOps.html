93          cloneCommand.setURI(repoURL);

102     /**
103      * @param repoDir The local directory in which to pull
104      * @param credentialsProvider credentialsProvider for the given repository.
105      * @return true if pull succeeded
106      * @throws IOException if an error occurs opening the local repository.
107      * @throws GitAPIException if an error occurs in a GIT API call.
108      */
109     boolean doPull(final File repoDir, final CredentialsProvider credentialsProvider) throws IOException, GitAPIException {
110         final Git git = Git.open(repoDir);
111 
112         final PullCommand pullCommand = git.pull();
113         pullCommand.setProgressMonitor(progressMonitor);
114         pullCommand.setCredentialsProvider(credentialsProvider);
115         final PullResult pullResult = pullCommand.call();
116         return pullResult.isSuccessful();
117     }

147     public List<String> cloneOrUpdateAll(final File workDir, final String user,
148                                          final String pwd) throws IOException {
149         createDir(workDir);
150 
151         final File userCloneDir = new File(workDir, user);
152         createDir(userCloneDir);
153 
154         final RepoList repoList = new RepoList(user, pwd);
155 
156         int count = 0;
157 
158         final List<User> orgs = repoList.getOrganizations();
159         System.out.println("Total Org count: " + orgs.size());
160 
161         final CredentialsProvider credentialsProvider
162                 = getCredentials(user, pwd);
163 
164         final List<String> failures = new ArrayList<String>();
165 
166         for (final User org : orgs) {
167             final List<Repository> orgRepos = repoList.getReposForOrg(org);
168             final File baseDirOrg = new File(userCloneDir, org.getLogin());
169             createDir(baseDirOrg);
170             System.out.println("Org: " + org.getLogin()
171                     + ", total org repos: " + orgRepos.size());
172 
173 
174             for (final Repository repo : orgRepos) {
175 
176                 final File repoDir = new File(baseDirOrg, repo.getName());
177                 cloneOrUpdate(credentialsProvider, failures, repo, repoDir);
178 
179                 count++;
180             }
181         }
182 
183 
184         final List<Repository> reposPublic = repoList.getReposPublic();
185         final File baseDirPublic = new File(userCloneDir, "public");
186         createDir(baseDirPublic);
187         System.out.println("Public repos: " + reposPublic.size());
188         for (Repository repo : reposPublic) {
189 
190             final File repoDir = new File(baseDirPublic, repo.getName());
191             cloneOrUpdate(credentialsProvider, failures, repo, repoDir);
192 
193             count++;
194         }
195 
196         System.out.println("Total Repo count: " + count);
197 
198         return failures;
199     }

201     /**
202      * @param credentialsProvider credentialsProvider for the given repo
203      * @param failures list of failures so far (will be appended to if needed).
204      * @param repo repository to clone or update (pull or fetch).
205      * @param repoDir local repository directory
206      * @throws IOException if an IO error occurs
207      */
208     void cloneOrUpdate(final CredentialsProvider credentialsProvider,
209                        final List<String> failures, final Repository repo,
210                        final File repoDir) throws IOException {
211 
212         System.out.print(repo.getCloneUrl());
213 
214         if (repoDir.exists()) {
215             System.out.print(" -- Pulling to: " + repoDir);
216             try {
217                 if (!doPull(repoDir, credentialsProvider)) {
218                     addFailurePull(failures, repo, repoDir, null);
219                 }
220             } catch (Exception e) {
221 
222                 System.out.print("\n\t\t  WARNING *** " + repo.getName()
223                         + " pull failed!, Fetching to: " + repoDir);
224                 try {
225                     doFetch(repoDir, credentialsProvider);
226                 } catch (Exception e1) {
227                     addFailurePull(failures, repo, repoDir, e);
228                 }
229 
230             }
231             System.out.println();
232 
233         } else {
234             System.out.println(" -- Cloning to: " + repoDir);
235             createDir(repoDir);
236 
237             try {
238                 doClone(repo.getCloneUrl(), repoDir,
239                         credentialsProvider);
240             } catch (Exception e) {
241                 addFailureClone(failures, repo, repoDir, e);
242             }
243         }
244     }

247      * @param failures list of failure messages, to which we will add the given
248      *                 error message and info.
249      * @param failedRepo the repository on which the failure occurred.
250      * @param repoDir the local directory containing the repository on which
251      *                the pull failed.
252      * @param cause the failure cause
253      */
254     void addFailurePull(final List<String> failures, final Repository failedRepo,
255                          final File repoDir, final Exception cause) {
256 
257         final String msgSuffix = (cause != null ? " -- message: " + cause.getMessage() : "");
258 
259         failures.add(failedRepo.getCloneUrl() + " -- " + repoDir + msgSuffix);
260 
261         System.out.println("Pull failed!!! repoDir: " + repoDir + msgSuffix);
262     }

264     /**
265      * @param failures list of failure messages, to which we will add the given
266      *                 error message and info.
267      * @param failedRepo the repository on which the failure occurred.
268      * @param repoCloneDir the local directory (which will be deleted)
269      *                     containing the failed repository clone.
270      * @param cause the failure cause
271      * @throws IOException if an IO error occurs
272      */
273     void addFailureClone(final List<String> failures, final Repository failedRepo,
274                          final File repoCloneDir, final Exception cause)
275             throws IOException {
276 
277         failures.add(failedRepo.getCloneUrl() + " -- " + repoCloneDir
278                 + " -- message: " + cause.getMessage());
279 
280         System.out.println("Clone failed!!! Deleting repoCloneDir: "
281                 + repoCloneDir + " -- message: " + cause.getMessage());
282 
283         // delete cloneDir, so subsequent attempts can succeed
284         delete(repoCloneDir);
285     }

287     /**
288      * Create all directories need for the given directory on disk.
289      * @param dir the dir tree to create.
290      * @throws IOException if an IO error occurs creating dirs.
291      */
292     static void createDir(final File dir) throws IOException {
293         if (!dir.exists() && !dir.mkdirs()) {
294             throw new IllegalStateException("Error creating directory: "
295                     + dir.getCanonicalPath());
296         }
297         if (!dir.isDirectory()) {
298             throw new IllegalStateException("Should be a directory: "
299                     + dir.getCanonicalPath());
300         }
301     }

308     static void delete(final File f) throws IOException {
309         if (f.isDirectory()) {
310             for (final File c : f.listFiles()) {
311                 delete(c);
312             }
313         }
314         if (!f.delete()) {
